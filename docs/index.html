Authentication Example

# Ali&Co FAPI 2.0 + DPoP Security Implementation
# Step 1: PAR (Pushed Authorization Request)
curl -X POST https://api.alico-banking.com/oauth2/par \
  -H "Content-Type: application/x-www-form-urlencoded" \
  -H "DPoP: ${DPOP_PROOF}" \
  -d "client_id=enterprise-banking-app" \
  -d "dpop_jkt=${JKT_THUMBPRINT}"

# Step 2: DPoP-bound Token Acquisition
curl -X POST https://api.alico-banking.com/oauth2/token \
  -H "DPoP: ${DPOP_PROOF}" \
  -d "grant_type=authorization_code" \
  -d "code=${AUTHORIZATION_CODE}"

# Step 3: Secure API Access with DPoP-bound Token
curl -X GET https://api.alico-banking.com/api/v1/customers/123 \
  -H "Authorization: DPoP ${DPOP_BOUND_TOKEN}" \
  -H "DPoP: ${DPOP_PROOF}" \
  -H "X-FAPI-Interaction-ID: $(uuidgen)"
